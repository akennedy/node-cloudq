queue job

param: name - Name of Queue param: job - Job Object param: cb - callback 

  queueJob: (name, job, cb) ->
    _.extend job, 
      queue: name
      queue_state: @QUEUED
      inserted_at: new Date()
    @jobs.insert job, cb

reserve job for processing

param: name - Name of Queue param: cb - Callback

 
  reserveJob: (name, cb) ->
    @jobs.findAndModify(
      {queue: name, queue_state: @QUEUED }
      , [['inserted_at', 'ascending']]
      , {$set: {queue_state: @RESERVED, updated_at: new Date() }}
      , {new: true }
      cb
    )
